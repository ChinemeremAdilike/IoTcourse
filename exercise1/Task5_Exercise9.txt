#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// LCD address is usually 0x27 in Wokwi
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  lcd.init();        // Initialize the LCD
  lcd.backlight();   // Turn on backlight

  lcd.setCursor(0, 0);
  lcd.print("I love singing..");
}

void loop() {
  lcd.setCursor(0, 1);
  lcd.print("Time: ");
  lcd.print(millis() / 1000);
}
